class Zombie:
    def __init__(self, name):
        self.name = name
        self.speed = 0

    def eat(self, Human):
        if Human._mobility >= 30:
            pass
        elif Human._noise >= 20:
            Human.eaten(self)

    def __repr__(self):
        return f'{self.name}, speed:{self.speed}'


class Human:
    def __init__(self, name):
        self.name = name
        self._mobility = 0
        self._noise = 0
        self.__life = 1

    def run(self):
        self._mobility += 5
        self._noise += 10

    def eaten(self, Zombie):
        self._mobility = 0
        self._noise = 0
        self.__life = 0

    def __str__(self):
        if self._mobility == 0:
            return f'{self.name}이 먹혔습니다'

    def __repr__(self):
        return f'{self.name} mobility: {self._mobility}, noise:{self._noise}'

    @property
    def mobility(self):
        return self._mobility

    @mobility.setter
    def new_mobility(self, new_mobility):
            self._mobility = new_mobility
            print(f'이동력을 {(new_mobility)}로 변경했습니다')
