```python
class Zombie:
    def __init__(self, name):
        self.name = name
        self.speed = 0

    def eat(self, Human):
        Human.eaten(self)

    def __repr__(self):
        return f'{self.name}, speed:{self.speed}'


class Human:
    def __init__(self, name):
        self.name = name
        self.mobility = 0
        self.noise = 0
        self.life = 1

    def run(self):
        self.mobility += 5
        self.noise += 10

    def eaten(self, Zombie):
        self.mobility = 0
        self.noise = 0
        self.life = 0

    def __str__(self):
        if self.mobility == 0:
            return f'{self.name}이 먹혔습니다'

    def __repr__(self):
        return f'{self.name} mobility: {self.mobility}, noise:{self.noise}'


Franken = Zombie('프랑켄')
Franken

Bob = Human('밥')
Bob.run()
Bob

Franken.eat(Bob)
Bob
```
